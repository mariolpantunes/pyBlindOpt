 44def scale(
 45    arr: np.ndarray,
 46    min_val: float | np.ndarray | None = None,
 47    max_val: float | np.ndarray | None = None,
 48) -> tuple[np.ndarray, float | np.ndarray, float | np.ndarray]:
 49    """
 50    Scales an array to the [0, 1] range using Min-Max scaling.
 51
 52    Scales an array to the range $[0, 1]$.
 53    $$ x_{scaled} = \frac{x - x_{min}}{x_{max} - x_{min}} $$
 54
 55    Args:
 56        arr (np.ndarray): The input array to scale.
 57        min_val (float | np.ndarray | None, optional): Minimum value for scaling. If None, computed from arr.
 58        max_val (float | np.ndarray | None, optional): Maximum value for scaling. If None, computed from arr.
 59
 60    Returns:
 61        tuple[np.ndarray, float | np.ndarray, float | np.ndarray]:
 62            - The scaled array.
 63            - The minimum value used.
 64            - The maximum value used.
 65    """
 66    # Use strict temporary variables to ensure type safety (guaranteed not None)
 67    actual_min = np.min(arr) if min_val is None else min_val
 68    actual_max = np.max(arr) if max_val is None else max_val
 69
 70    # Avoid division by zero if max == min
 71    denominator = actual_max - actual_min
 72
 73    if np.any(denominator == 0):
 74        scl_arr = np.zeros_like(arr)
 75    else:
 76        scl_arr = (arr - actual_min) / denominator
 77
 78    return scl_arr, actual_min, actual_max

 81def inv_scale(
 82    scl_arr: np.ndarray, min_val: float | np.ndarray, max_val: float | np.ndarray
 83) -> np.ndarray:
 84    """
 85    Inverse scales an array from [0, 1] back to the original range.
 86
 87    Restores values from $[0, 1]$ to $[x_{min}, x_{max}]$.
 88
 89    Args:
 90        scl_arr (np.ndarray): The scaled array.
 91        min_val (float | np.ndarray): The minimum value used in the original scaling.
 92        max_val (float | np.ndarray): The maximum value used in the original scaling.
 93
 94    Returns:
 95        np.ndarray: The array rescaled to the original range.
 96    """
 97    return scl_arr * (max_val - min_val) + min_val

100class Sampler(abc.ABC):
101    """
102    Abstract Base Class for Sampling Strategies.
103
104    Defines the interface for generating random or quasi-random numbers
105    within the search space.
106    """
107
108    def __init__(self, rng: np.random.Generator):
109        """
110        Args:
111            rng (np.random.Generator): The centralized random number generator.
112        """
113        self.rng = rng
114
115    @abc.abstractmethod
116    def sample(self, n_pop: int, bounds: np.ndarray) -> np.ndarray:
117        """
118        Generates $N$ samples within the given bounds.
119
120        Args:
121            n_pop (int): Number of individuals.
122            bounds (np.ndarray): Search space bounds (shape: D x 2).
123
124        Returns:
125            np.ndarray: Population matrix of shape (n_pop, D).
126        """
127        pass
128
129    def _scale_to_bounds(
130        self, unit_samples: np.ndarray, bounds: np.ndarray
131    ) -> np.ndarray:
132        """
133        Helper to scale [0, 1] samples to [min, max] bounds.
134        """
135        min_b = bounds[:, 0]
136        max_b = bounds[:, 1]
137        return inv_scale(unit_samples, min_b, max_b)

154class HLCSampler(Sampler):
155    """
156        Hyper-Latin Cube Sampling (LHS).
157
158    Stratified sampling that ensures coverage across all dimensions.
159    Divides each dimension into $N$ intervals and places exactly one sample per interval,
160    minimizing clustering.
161    """
162
163    def sample(self, n_pop: int, bounds: np.ndarray) -> np.ndarray:
164        dim = bounds.shape[0]
165        # 1. Generate stratified samples in [0, 1]
166        samples = np.zeros((dim, n_pop))
167
168        # Divide [0,1] into n_pop intervals
169        step = 1.0 / n_pop
170
171        for d in range(dim):
172            # Create points: [0, 1/N, 2/N, ...]
173            points = np.arange(n_pop) * step
174
175            # Add random jitter within each interval
176            jitter = self.rng.uniform(0, step, size=n_pop)
177            points += jitter
178
179            # Shuffle this dimension independently so dimensions are uncorrelated
180            self.rng.shuffle(points)
181            samples[d] = points
182
183        # Transpose to (N, D) and scale
184        return self._scale_to_bounds(samples.T, bounds)

187class SobolSampler(Sampler):
188    """
189    Sobol Sequence Sampler.
190
191    A low-discrepancy quasi-random sequence. It fills the space more evenly than
192    random sampling, reducing gaps and clusters.
193
194    **Implementation:**
195    Pure NumPy implementation using Gray Code and pre-computed direction numbers.
196    Supports up to 40 dimensions (Joe & Kuo, 2003).
197    """
198
199    # Format: [d, s, a, m_i...]
200    # d: dimension index
201    # s: degree of primitive polynomial
202    # a: polynomial coefficient (integer representing the polynomial)
203    # m: initial direction numbers
204    # Source: Joe & Kuo (2003)
205    _DIRECTION_NUMBERS = [
206        # Dim 1 (skipped, handled as special case)
207        # Dim 2-10
208        [2, 1, 0, [1]],
209        [3, 2, 1, [1, 3]],
210        [4, 3, 1, [1, 3, 1]],
211        [5, 3, 2, [1, 1, 1]],
212        [6, 4, 1, [1, 1, 3, 3]],
213        [7, 4, 4, [1, 3, 5, 13]],
214        [8, 5, 2, [1, 1, 5, 5, 17]],
215        [9, 5, 4, [1, 1, 5, 5, 5]],
216        [10, 5, 7, [1, 1, 7, 11, 19]],
217        # Dim 11-20
218        [11, 5, 11, [1, 1, 7, 13, 25]],
219        [12, 5, 13, [1, 1, 5, 11, 25]],
220        [13, 5, 14, [1, 1, 3, 13, 27]],
221        [14, 6, 1, [1, 1, 1, 3, 11, 25]],
222        [15, 6, 13, [1, 3, 1, 13, 27, 43]],
223        [16, 6, 16, [1, 1, 5, 5, 29, 39]],
224        [17, 6, 19, [1, 1, 7, 7, 21, 37]],
225        [18, 6, 22, [1, 1, 1, 9, 23, 37]],
226        [19, 6, 25, [1, 1, 3, 13, 31, 11]],
227        [20, 6, 1, [1, 3, 3, 9, 9, 57]],
228        # Dim 21-30
229        [21, 6, 4, [1, 3, 7, 13, 29, 19]],
230        [22, 7, 1, [1, 1, 1, 1, 3, 15, 29]],
231        [23, 7, 2, [1, 1, 5, 11, 27, 27, 57]],
232        [24, 7, 1, [1, 3, 5, 15, 5, 29, 43]],
233        [25, 7, 13, [1, 3, 1, 1, 23, 37, 65]],
234        [26, 7, 16, [1, 1, 3, 3, 13, 5, 87]],
235        [27, 7, 19, [1, 1, 5, 13, 7, 43, 9]],
236        [28, 7, 22, [1, 1, 7, 9, 15, 11, 21]],
237        [29, 7, 1, [1, 3, 1, 5, 1, 25, 71]],
238        [30, 7, 1, [1, 1, 3, 15, 11, 55, 35]],
239        # Dim 31-40
240        [31, 7, 4, [1, 1, 1, 11, 21, 17, 105]],
241        [32, 7, 4, [1, 3, 5, 3, 7, 25, 61]],
242        [33, 7, 7, [1, 3, 1, 1, 29, 17, 111]],
243        [34, 7, 7, [1, 1, 5, 9, 19, 53, 59]],
244        [35, 7, 7, [1, 1, 3, 3, 11, 63, 13]],
245        [36, 7, 19, [1, 1, 7, 5, 23, 49, 101]],
246        [37, 7, 19, [1, 1, 1, 7, 5, 17, 77]],
247        [38, 7, 21, [1, 1, 5, 15, 27, 5, 89]],
248        [39, 7, 21, [1, 3, 3, 9, 21, 15, 31]],
249        [40, 7, 21, [1, 3, 5, 13, 7, 39, 27]],
250    ]

252    def _compute_v(self, dim_idx):
253        """Computes direction numbers V for a specific dimension."""
254        BITS = 32
255
256        # 1. Handle Dimension 1 (Special Case: Van der Corput)
257        if dim_idx == 0:
258            V = np.zeros(BITS + 1, dtype=np.uint32)
259            for i in range(1, BITS + 1):
260                V[i] = 1 << (BITS - i)
261            return V
262
263        # 2. Handle Dimensions 2+
264        if dim_idx > len(self._DIRECTION_NUMBERS):
265            raise ValueError(
266                f"Max dimension supported is {len(self._DIRECTION_NUMBERS) + 1}"
267            )
268
269        params = self._DIRECTION_NUMBERS[dim_idx - 1]
270        s = params[1]
271        a = params[2]
272        m = [0] + params[3]
273
274        V = np.zeros(BITS + 1, dtype=np.uint32)
275
276        # Initialize first 's' numbers
277        for i in range(1, s + 1):
278            V[i] = m[i] << (BITS - i)
279
280        # Recurrence for remaining bits
281        for i in range(s + 1, BITS + 1):
282            v_new = V[i - s] ^ (V[i - s] >> s)
283            for k in range(1, s):
284                if (a >> (s - 1 - k)) & 1:
285                    v_new ^= V[i - k]
286            V[i] = v_new
287        return V
288
289    def sample(self, n_pop: int, bounds: np.ndarray) -> np.ndarray:
290        dim = bounds.shape[0]
291
292        if dim > len(self._DIRECTION_NUMBERS) + 1:
293            raise ValueError(
294                f"Requested {dim} dimensions, max is {len(self._DIRECTION_NUMBERS) + 1}"
295            )
296
297        BITS = 32
298        SCALE = 2**BITS
299
300        # 1. Compute V table
301        V = np.zeros((dim, BITS + 1), dtype=np.uint32)
302        for d in range(dim):
303            V[d] = self._compute_v(d)
304
305        # 2. Generate Points (Gray Code)
306        samples_int = np.zeros((n_pop, dim), dtype=np.uint32)
307        X = np.zeros(dim, dtype=np.uint32)
308
309        # Simplified scrambling
310        scramble = self.rng.integers(0, SCALE, size=dim, dtype=np.uint32)
311
312        for i in range(n_pop):
313            # Find index of rightmost zero bit (equivalent to rightmost set bit of ~i)
314            # This 'c' tells us which Direction Number to XOR
315            # i=0 (..00) -> c=1
316            # i=1 (..01) -> c=2
317            c = 1
318            value = i
319            while value & 1:
320                value >>= 1
321                c += 1
322
323            if c < BITS:
324                X ^= V[:, c]
325
326            samples_int[i] = X ^ scramble
327
328        return self._scale_to_bounds(samples_int / float(SCALE), bounds)

331class ChaoticSampler(Sampler):
332    """
333    Chaotic Map Sampling (Logistic Map).
334
335    Uses a deterministic chaotic system to generate samples.
336    $$ x_{k+1} = r \\cdot x_k (1 - x_k) $$
337    where $r=4.0$.
338    Chaos is ergodic and can provide better exploration dynamics.
339    """
340
341    def sample(self, n_pop: int, bounds: np.ndarray) -> np.ndarray:
342        dim = bounds.shape[0]
343        # Standard Logistic Map parameter for chaos
344        r = 4.0
345
346        # Initialize with random start (avoid 0.0, 0.5, 1.0 fixed points)
347        # Shape (D,) one sequence per dimension
348        x = self.rng.uniform(0.1, 0.9, size=dim)
349
350        # Burn-in: Run map for 100 iterations to decouple from seed
351        for _ in range(100):
352            x = r * x * (1.0 - x)
353
354        # Generate samples
355        samples = np.zeros((n_pop, dim))
356        for i in range(n_pop):
357            x = r * x * (1.0 - x)
358            samples[i] = x
359
360        # Chaotic maps are naturally in [0, 1]
361        return self._scale_to_bounds(samples, bounds)

364def assert_bounds(solution: np.ndarray, bounds: np.ndarray) -> bool:
365    """
366    Verifies if the solution is contained within the defined bounds.
367
368    Args:
369        solution (np.ndarray): The solution vector(s) to check.
370        bounds (np.ndarray): The bounds of valid solutions (shape: N x 2).
371
372    Returns:
373        bool: True if the solution is within bounds, False otherwise.
374    """
375    x = solution.T
376    min_bounds = bounds[:, 0]
377    max_bounds = bounds[:, 1]
378    rv = ((x > min_bounds[:, np.newaxis]) & (x < max_bounds[:, np.newaxis])).any(1)
379    return bool(np.all(rv))

415def global_distances(samples: np.ndarray) -> np.ndarray:
416    """
417    Computes global Euclidean distance sum.
418
419    Calculates $\\sum_j ||x_i - x_j||$ for every sample $i$.
420    Used to measure isolation/centrality.
421
422    Args:
423        samples (np.ndarray): Shape (N, D).
424
425    Returns:
426        np.ndarray: Shape (N,). The sum of distances for each sample.
427    """
428    # 1. Compute Pairwise Differences
429    diff = samples[:, np.newaxis, :] - samples[np.newaxis, :, :]
430
431    # 2. Euclidean Distance Matrix
432    sq_dist = np.sum(diff**2, axis=-1)
433    dist_matrix = np.sqrt(sq_dist)
434
435    # 3. Sum rows to get total distance to all others
436    return np.sum(dist_matrix, axis=1)

439def compute_crowding_distance(samples: np.ndarray) -> np.ndarray:
440    """
441    Crowding Distance Calculation (NSGA-II).
442
443    Estimates the density of solutions surrounding a particular point in the objective space.
444    Higher distance = More isolated (Better for diversity).
445
446    Args:
447        samples (np.ndarray): Shape (N, D)
448
449    Returns:
450        np.ndarray: Shape (N,). Higher value = More isolated (Better).
451    """
452    N, D = samples.shape
453    if N == 0:
454        return np.array([])
455
456    distances = np.zeros(N)
457
458    # We compute distance dimension by dimension
459    for d in range(D):
460        # 1. Sort by the current dimension
461        # argsort gives us the indices that would sort the array
462        sorted_indices = np.argsort(samples[:, d])
463        sorted_samples = samples[sorted_indices, d]
464
465        # 2. Handle Boundaries
466        # The min and max points in each dim are always "infinite" distance (most isolated)
467        distances[sorted_indices[0]] = np.inf
468        distances[sorted_indices[-1]] = np.inf
469
470        # 3. Compute Distance for Internal Points
471        # Formula: (Next_Val - Prev_Val) / (Max_Val - Min_Val)
472        scale = sorted_samples[-1] - sorted_samples[0]
473
474        if scale == 0:
475            continue  # All points are identical in this dimension
476
477        # Vectorized difference: P[i+1] - P[i-1]
478        # We slice sorted_samples from [2:] and [:-2] to get next/prev neighbors
479        dim_dist = (sorted_samples[2:] - sorted_samples[:-2]) / scale
480
481        # Add to the cumulative score of the corresponding original indices
482        # Indices [1:-1] correspond to the internal points we just computed
483        distances[sorted_indices[1:-1]] += dim_dist
484
485    # Replace infinite values (boundaries) with the max finite value found
486    # so probabilities don't break
487    finite_mask = np.isfinite(distances)
488    if np.any(finite_mask):
489        max_dist = np.max(distances[finite_mask])
490        distances[~finite_mask] = max_dist * 2.0
491    else:
492        # If all are infinite (e.g. N=2), fallback to ones
493        distances[:] = 1.0
494
495    return distances

498def score_2_probs(scores: np.ndarray, temperature: float = 1.0) -> np.ndarray:
499    """
500    Softmax conversion of scores to probabilities.
501
502    Converts minimization costs into selection probabilities.
503    $$ P(i) = \\frac{\\exp(-score_i / T)}{\\sum \\exp(-score_j / T)} $$
504
505    Args:
506        scores: Cost values (lower is better).
507        temperature:
508             < 1.0: Sharper distribution (Greedy).
509             = 1.0: Standard Boltzmann.
510             > 1.0: Flatter distribution (Random).
511    """
512    # 1. Check for flat scores (avoid division by zero later)
513    if np.all(scores == scores[0]):
514        return np.ones_like(scores) / len(scores)
515
516    # 2. Softmax Logic
517    # We negate scores because Softmax maximizes, but we want to minimize cost.
518    neg_scores = -scores
519
520    # 3. Numerical Stability (Log-Sum-Exp trick equivalent)
521    # Subtract max(neg_scores) so the largest exponent is 0.
522    # (This corresponds to subtracting the *minimum* original score).
523    shift = np.max(neg_scores)
524
525    # Apply Temperature and Shift
526    # Avoid T=0 crash
527    temp = max(temperature, 1e-9)
528    exps = np.exp((neg_scores - shift) / temp)
529
530    # 4. Normalize
531    probs = exps / np.sum(exps)
532
533    # 5. Safety clamp (fix floating point slight errors < 0 or > 1)
534    probs = np.clip(probs, 0.0, 1.0)
535    return probs / np.sum(probs)

538def compute_objective(
539    population: np.ndarray, function: Callable[[object], float], n_jobs: int = 1
540) -> np.ndarray:
541    """
542    Computes the objective function for a population of solutions.
543
544    Strategy:
545    1. Optimistic Vectorization: Tries passing the entire population matrix to the function.
546    2. Serial (n_jobs=1): Uses np.apply_along_axis for row-wise evaluation.
547    3. Parallel (n_jobs!=1): Uses Joblib for multiprocessing.
548
549    Args:
550        population (np.ndarray): The population of solutions to evaluate.
551        function (Callable[[object], float]): The objective function to apply.
552        n_jobs (int, optional): Number of parallel jobs. 1 forces serial. Defaults to 1.
553
554    Returns:
555        np.ndarray: A NumPy array of objective values.
556    """
557    # Ensure input is a standard numpy array for consistent handling
558    if isinstance(population, list):
559        population = np.array(population)
560
561    # 1. Optimistic Approach: Vectorized Execution
562    # If the user's function supports (N, D) -> (N,) input, this is instant.
563    try:
564        result = function(population)
565        # Verify result is a valid array of the correct shape (N,) or (N, 1)
566        if isinstance(result, np.ndarray) and result.size == population.shape[0]:
567            return result.flatten()
568    except Exception:
569        # Function does not support matrix input, proceed to row-by-row methods
570        pass
571
572    # 2. Serial Execution (User requested np.apply_along_axis)
573    if n_jobs == 1:
574        # Apply function along axis 1 (rows).
575        # Note: apply_along_axis iterates in Python but handles array wrapping cleanly.
576        return np.apply_along_axis(function, 1, population)
577
578    # 3. Parallel Execution (Joblib)
579    else:
580        try:
581            # Backend 'loky' is robust for generic Python objects.
582            obj_list = joblib.Parallel(backend="loky", n_jobs=n_jobs)(
583                joblib.delayed(function)(c) for c in population
584            )
585        except Exception:
586            # Fallback to threading if serialization (pickling) fails
587            obj_list = joblib.Parallel(backend="threading", n_jobs=n_jobs)(
588                joblib.delayed(function)(c) for c in population
589            )
590
591        return np.array(obj_list)

594def levy_flight(
595    n_rows: int, n_cols: int, beta: float = 1.5, rng: np.random.Generator | None = None
596) -> np.ndarray:
597    """
598    LÃ©vy Flight Step Generation.
599
600    Generates steps from a heavy-tailed distribution (LÃ©vy distribution),
601    simulating the flight patterns of foraging animals (short steps + rare long jumps).
602
603    **Mantegna's Algorithm:**
604    $$ \\text{Step} = \\frac{u}{|v|^{1/\\beta}} $$
605    where $u \\sim \\mathcal{N}(0, \\sigma_u^2)$ and $v \\sim \\mathcal{N}(0, 1)$.
606    """
607    if rng is None:
608        rng = np.random.default_rng()
609
610    sigma_u = (
611        math.gamma(1 + beta)
612        * math.sin(math.pi * beta / 2)
613        / (math.gamma((1 + beta) / 2) * beta * 2 ** ((beta - 1) / 2))
614    ) ** (1 / beta)
615
616    u = rng.normal(0, sigma_u, size=(n_rows, n_cols))
617    v = rng.normal(0, 1, size=(n_rows, n_cols))
618
619    # Avoid division by zero
620    v[v == 0] = 1e-10
621
622    step = u / (np.abs(v) ** (1 / beta))
623    return step

253    def _compute_v(self, dim_idx):
254        """Computes direction numbers V for a specific dimension."""
255        BITS = 32
256
257        # 1. Handle Dimension 1 (Special Case: Van der Corput)
258        if dim_idx == 0:
259            V = np.zeros(BITS + 1, dtype=np.uint32)
260            for i in range(1, BITS + 1):
261                V[i] = 1 << (BITS - i)
262            return V
263
264        # 2. Handle Dimensions 2+
265        if dim_idx > len(self._DIRECTION_NUMBERS):
266            raise ValueError(
267                f"Max dimension supported is {len(self._DIRECTION_NUMBERS) + 1}"
268            )
269
270        params = self._DIRECTION_NUMBERS[dim_idx - 1]
271        s = params[1]
272        a = params[2]
273        m = [0] + params[3]
274
275        V = np.zeros(BITS + 1, dtype=np.uint32)
276
277        # Initialize first 's' numbers
278        for i in range(1, s + 1):
279            V[i] = m[i] << (BITS - i)
280
281        # Recurrence for remaining bits
282        for i in range(s + 1, BITS + 1):
283            v_new = V[i - s] ^ (V[i - s] >> s)
284            for k in range(1, s):
285                if (a >> (s - 1 - k)) & 1:
286                    v_new ^= V[i - k]
287            V[i] = v_new
288        return V

291        dim = bounds.shape[0]
292
293        if dim > len(self._DIRECTION_NUMBERS) + 1:
294            raise ValueError(
295                f"Requested {dim} dimensions, max is {len(self._DIRECTION_NUMBERS) + 1}"
296            )
297
298        BITS = 32
299        SCALE = 2**BITS
300
301        # 1. Compute V table
302        V = np.zeros((dim, BITS + 1), dtype=np.uint32)
303        for d in range(dim):
304            V[d] = self._compute_v(d)
305
306        # 2. Generate Points (Gray Code)
307        samples_int = np.zeros((n_pop, dim), dtype=np.uint32)
308        X = np.zeros(dim, dtype=np.uint32)
309
310        # Simplified scrambling
311        scramble = self.rng.integers(0, SCALE, size=dim, dtype=np.uint32)
312
313        for i in range(n_pop):
314            # Find index of rightmost zero bit (equivalent to rightmost set bit of ~i)
315            # This 'c' tells us which Direction Number to XOR
316            # i=0 (..00) -> c=1
317            # i=1 (..01) -> c=2
318            c = 1
319            value = i
320            while value & 1:
321                value >>= 1
322                c += 1
323
324            if c < BITS:
325                X ^= V[:, c]
326
327            samples_int[i] = X ^ scramble
328
329        return self._scale_to_bounds(samples_int / float(SCALE), bounds)

def compute_crowding_distance(samples: numpy.ndarray) -> numpy.ndarray: 
440def compute_crowding_distance(samples: np.ndarray) -> np.ndarray:
441    """
442    Crowding Distance Calculation (NSGA-II).
443
444    Estimates the density of solutions surrounding a particular point in the objective space.
445    Higher distance = More isolated (Better for diversity).
446
447    Args:
448        samples (np.ndarray): Shape (N, D)
449
450    Returns:
451        np.ndarray: Shape (N,). Higher value = More isolated (Better).
452    """
453    N, D = samples.shape
454    if N == 0:
455        return np.array([])
456
457    distances = np.zeros(N)
458
459    # We compute distance dimension by dimension
460    for d in range(D):
461        # 1. Sort by the current dimension
462        # argsort gives us the indices that would sort the array
463        sorted_indices = np.argsort(samples[:, d])
464        sorted_samples = samples[sorted_indices, d]
465
466        # 2. Handle Boundaries
467        # The min and max points in each dim are always "infinite" distance (most isolated)
468        distances[sorted_indices[0]] = np.inf
469        distances[sorted_indices[-1]] = np.inf
470
471        # 3. Compute Distance for Internal Points
472        # Formula: (Next_Val - Prev_Val) / (Max_Val - Min_Val)
473        scale = sorted_samples[-1] - sorted_samples[0]
474
475        if scale == 0:
476            continue  # All points are identical in this dimension
477
478        # Vectorized difference: P[i+1] - P[i-1]
479        # We slice sorted_samples from [2:] and [:-2] to get next/prev neighbors
480        dim_dist = (sorted_samples[2:] - sorted_samples[:-2]) / scale
481
482        # Add to the cumulative score of the corresponding original indices
483        # Indices [1:-1] correspond to the internal points we just computed
484        distances[sorted_indices[1:-1]] += dim_dist
485
486    # Replace infinite values (boundaries) with the max finite value found
487    # so probabilities don't break
488    finite_mask = np.isfinite(distances)
489    if np.any(finite_mask):
490        max_dist = np.max(distances[finite_mask])
491        distances[~finite_mask] = max_dist * 2.0
492    else:
493        # If all are infinite (e.g. N=2), fallback to ones
494        distances[:] = 1.0
495
496    return distances

def score_2_probs(scores: numpy.ndarray, temperature: float = 1.0) -> numpy.ndarray: 
499def score_2_probs(scores: np.ndarray, temperature: float = 1.0) -> np.ndarray:
500    """
501    Softmax conversion of scores to probabilities.
502
503    Converts minimization costs into selection probabilities.
504    $$ P(i) = \\frac{\\exp(-score_i / T)}{\\sum \\exp(-score_j / T)} $$
505
506    Args:
507        scores: Cost values (lower is better).
508        temperature:
509             < 1.0: Sharper distribution (Greedy).
510             = 1.0: Standard Boltzmann.
511             > 1.0: Flatter distribution (Random).
512    """
513    # 1. Check for flat scores (avoid division by zero later)
514    if np.all(scores == scores[0]):
515        return np.ones_like(scores) / len(scores)
516
517    # 2. Softmax Logic
518    # We negate scores because Softmax maximizes, but we want to minimize cost.
519    neg_scores = -scores
520
521    # 3. Numerical Stability (Log-Sum-Exp trick equivalent)
522    # Subtract max(neg_scores) so the largest exponent is 0.
523    # (This corresponds to subtracting the *minimum* original score).
524    shift = np.max(neg_scores)
525
526    # Apply Temperature and Shift
527    # Avoid T=0 crash
528    temp = max(temperature, 1e-9)
529    exps = np.exp((neg_scores - shift) / temp)
530
531    # 4. Normalize
532    probs = exps / np.sum(exps)
533
534    # 5. Safety clamp (fix floating point slight errors < 0 or > 1)
535    probs = np.clip(probs, 0.0, 1.0)
536    return probs / np.sum(probs)

579    # 3. Parallel Execution (Joblib)

 63class HLCSampler(Sampler):
 64    """
 65    Hyper-Latin Cube Sampling (LHS).
 66    Ensures stratified sampling across all dimensions.
 67    """
 68    def sample(self, n_pop: int, bounds: np.ndarray) -> np.ndarray:
 69        dim = bounds.shape[0]
 70        # 1. Generate stratified samples in [0, 1]
 71        samples = np.zeros((dim, n_pop))
 72        
 73        # Divide [0,1] into n_pop intervals
 74        step = 1.0 / n_pop
 75        
 76        for d in range(dim):
 77            # Create points: [0, 1/N, 2/N, ...]
 78            points = np.arange(n_pop) * step
 79            
 80            # Add random jitter within each interval
 81            jitter = self.rng.uniform(0, step, size=n_pop)
 82            points += jitter
 83            
 84            # Shuffle this dimension independently so dimensions are uncorrelated
 85            self.rng.shuffle(points)
 86            samples[d] = points
 87            
 88        # Transpose to (N, D) and scale
 89        return self._scale_to_bounds(samples.T, bounds)

152    def _compute_v(self, dim_idx):
153        """Computes direction numbers V for a specific dimension."""
154        BITS = 32
155        
156        # 1. Handle Dimension 1 (Special Case: Van der Corput)
157        if dim_idx == 0:
158            V = np.zeros(BITS + 1, dtype=np.uint32)
159            for i in range(1, BITS + 1):
160                V[i] = 1 << (BITS - i)
161            return V
162
163        # 2. Handle Dimensions 2+
164        if dim_idx > len(self._DIRECTION_NUMBERS):
165            raise ValueError(f"Max dimension supported is {len(self._DIRECTION_NUMBERS)+1}")
166
167        params = self._DIRECTION_NUMBERS[dim_idx - 1]
168        s = params[1]
169        a = params[2]
170        m = [0] + params[3]
171
172        V = np.zeros(BITS + 1, dtype=np.uint32)
173
174        # Initialize first 's' numbers
175        for i in range(1, s + 1):
176            V[i] = m[i] << (BITS - i)
177
178        # Recurrence for remaining bits
179        for i in range(s + 1, BITS + 1):
180            v_new = V[i - s] ^ (V[i - s] >> s)
181            for k in range(1, s):
182                if (a >> (s - 1 - k)) & 1:
183                    v_new ^= V[i - k]
184            V[i] = v_new
185        return V
186
187    def sample(self, n_pop: int, bounds: np.ndarray) -> np.ndarray:
188        dim = bounds.shape[0]
189        
190        if dim > len(self._DIRECTION_NUMBERS) + 1:
191            raise ValueError(f"Requested {dim} dimensions, max is {len(self._DIRECTION_NUMBERS)+1}")
192
193        BITS = 32
194        SCALE = 2**BITS
195        
196        # 1. Compute V table
197        V = np.zeros((dim, BITS + 1), dtype=np.uint32)
198        for d in range(dim):
199            V[d] = self._compute_v(d)
200
201        # 2. Generate Points (Gray Code)
202        samples_int = np.zeros((n_pop, dim), dtype=np.uint32)
203        X = np.zeros(dim, dtype=np.uint32)
204        
205        # Simplified scrambling
206        scramble = self.rng.integers(0, SCALE, size=dim, dtype=np.uint32)
207        
208        for i in range(n_pop):
209            # Find index of rightmost zero bit (equivalent to rightmost set bit of ~i)
210            # This 'c' tells us which Direction Number to XOR
211            # i=0 (..00) -> c=1
212            # i=1 (..01) -> c=2
213            c = 1
214            value = i
215            while value & 1:
216                value >>= 1
217                c += 1
218            
219            if c < BITS:
220                X ^= V[:, c]
221            
222            samples_int[i] = X ^ scramble
223
224        return self._scale_to_bounds(samples_int / float(SCALE), bounds)

227def assert_bounds(solution: np.ndarray, bounds: np.ndarray) -> bool:
228    """
229    Verifies if the solution is contained within the defined bounds.
230
231    Args:
232        solution (np.ndarray): The solution vector(s) to check.
233        bounds (np.ndarray): The bounds of valid solutions (shape: N x 2).
234
235    Returns:
236        bool: True if the solution is within bounds, False otherwise.
237    """
238    x = solution.T
239    min_bounds = bounds[:, 0]
240    max_bounds = bounds[:, 1]
241    rv = ((x > min_bounds[:, np.newaxis]) & (x < max_bounds[:, np.newaxis])).any(1)
242    return bool(np.all(rv))

276def scale(
277    arr: np.ndarray,
278    min_val: float | np.ndarray | None = None,
279    max_val: float | np.ndarray | None = None,
280) -> tuple[np.ndarray, float | np.ndarray, float | np.ndarray]:
281    """
282    Scales an array to the [0, 1] range using Min-Max scaling.
283
284    Args:
285        arr (np.ndarray): The input array to scale.
286        min_val (float | np.ndarray | None, optional): Minimum value for scaling. If None, computed from arr.
287        max_val (float | np.ndarray | None, optional): Maximum value for scaling. If None, computed from arr.
288
289    Returns:
290        tuple[np.ndarray, float | np.ndarray, float | np.ndarray]:
291            - The scaled array.
292            - The minimum value used.
293            - The maximum value used.
294    """
295    # Use strict temporary variables to ensure type safety (guaranteed not None)
296    actual_min = np.min(arr) if min_val is None else min_val
297    actual_max = np.max(arr) if max_val is None else max_val
298
299    # Avoid division by zero if max == min
300    denominator = actual_max - actual_min
301
302    if np.any(denominator == 0):
303        scl_arr = np.zeros_like(arr)
304    else:
305        scl_arr = (arr - actual_min) / denominator
306
307    return scl_arr, actual_min, actual_max

310def inv_scale(
311    scl_arr: np.ndarray, min_val: float | np.ndarray, max_val: float | np.ndarray
312) -> np.ndarray:
313    """
314    Inverse scales an array from [0, 1] back to the original range.
315
316    Args:
317        scl_arr (np.ndarray): The scaled array.
318        min_val (float | np.ndarray): The minimum value used in the original scaling.
319        max_val (float | np.ndarray): The maximum value used in the original scaling.
320
321    Returns:
322        np.ndarray: The array rescaled to the original range.
323    """
324    return scl_arr * (max_val - min_val) + min_val

327def global_distances(samples: np.ndarray) -> np.ndarray:
328    """
329    Computes the sum of Euclidean distances from each sample to every other sample.
330    Vectorized implementation O(N^2).
331
332    Args:
333        samples (np.ndarray): Shape (N, D).
334
335    Returns:
336        np.ndarray: Shape (N,). The sum of distances for each sample.
337    """
338    # 1. Compute Pairwise Differences
339    diff = samples[:, np.newaxis, :] - samples[np.newaxis, :, :]
340
341    # 2. Euclidean Distance Matrix
342    sq_dist = np.sum(diff**2, axis=-1)
343    dist_matrix = np.sqrt(sq_dist)
344
345    # 3. Sum rows to get total distance to all others
346    return np.sum(dist_matrix, axis=1)

349def compute_crowding_distance(samples: np.ndarray) -> np.ndarray:
350    """
351    Computes the Crowding Distance (NSGA-II style).
352    Estimates the density of points around each sample.
353    
354    Complexity: O(D * N log N)
355    
356    Args:
357        samples (np.ndarray): Shape (N, D)
358        
359    Returns:
360        np.ndarray: Shape (N,). Higher value = More isolated (Better).
361    """
362    N, D = samples.shape
363    if N == 0:
364        return np.array([])
365
366    distances = np.zeros(N)
367
368    # We compute distance dimension by dimension
369    for d in range(D):
370        # 1. Sort by the current dimension
371        # argsort gives us the indices that would sort the array
372        sorted_indices = np.argsort(samples[:, d])
373        sorted_samples = samples[sorted_indices, d]
374
375        # 2. Handle Boundaries
376        # The min and max points in each dim are always "infinite" distance (most isolated)
377        distances[sorted_indices[0]] = np.inf
378        distances[sorted_indices[-1]] = np.inf
379
380        # 3. Compute Distance for Internal Points
381        # Formula: (Next_Val - Prev_Val) / (Max_Val - Min_Val)
382        scale = sorted_samples[-1] - sorted_samples[0]
383        
384        if scale == 0:
385            continue  # All points are identical in this dimension
386
387        # Vectorized difference: P[i+1] - P[i-1]
388        # We slice sorted_samples from [2:] and [:-2] to get next/prev neighbors
389        dim_dist = (sorted_samples[2:] - sorted_samples[:-2]) / scale
390        
391        # Add to the cumulative score of the corresponding original indices
392        # Indices [1:-1] correspond to the internal points we just computed
393        distances[sorted_indices[1:-1]] += dim_dist
394
395    # Replace infinite values (boundaries) with the max finite value found 
396    # so probabilities don't break
397    max_finite = np.max(distances[np.isfinite(distances)])
398    distances[np.isinf(distances)] = max_finite * 2.0  # Give them a boost
399
400    return distances

403def score_2_probs(scores: np.ndarray, temperature: float = 1.0) -> np.ndarray:
404    """
405    Converts minimization scores into a probability distribution.
406    Uses Min-Max normalization to handle negative values robustly.
407
408    Args:
409        scores (np.ndarray): Objective values (lower is better).
410        temperature (float): Controls selection pressure (optional). 
411                             >1.0 makes distribution flatter (more random).
412                             <1.0 makes distribution sharper (greedy).
413
414    Returns:
415        np.ndarray: Probabilities summing to 1.0.
416    """
417    # 1. Handle Constant Scores (Variance is 0)
418    if np.all(scores == scores[0]):
419        return np.ones_like(scores) / len(scores)
420
421    # 2. Min-Max Normalization -> Maps to [0, 1]
422    # S_norm = (S - min) / (max - min)
423    min_s = np.min(scores)
424    max_s = np.max(scores)
425    norm_scores, _, _ = scale(scores, min_s, max_s)
426
427    # 3. Invert (Minimization: Low score -> High prob)
428    # 0.0 (Best) becomes 1.0
429    # 1.0 (Worst) becomes 0.0
430    # We add a small epsilon to avoid 0.0 probability for the worst candidate
431    weights = (1.0 - norm_scores) + 1e-6
432
433    # 4. Optional: Temperature Scaling (Weights ^ (1/T))
434    if temperature != 1.0:
435        weights = weights ** (1.0 / temperature)
436
437    # 5. Normalize to Probabilities
438    return weights / np.sum(weights)

441def compute_objective(
442    population: np.ndarray, function: Callable[[object], float], n_jobs: int = 1
443) -> np.ndarray:
444    """
445    Computes the objective function for a population of solutions.
446
447    Strategy:
448    1. Optimistic Vectorization: Tries passing the entire population matrix to the function.
449    2. Serial (n_jobs=1): Uses np.apply_along_axis for row-wise evaluation.
450    3. Parallel (n_jobs!=1): Uses Joblib for multiprocessing.
451
452    Args:
453        population (np.ndarray): The population of solutions to evaluate.
454        function (Callable[[object], float]): The objective function to apply.
455        n_jobs (int, optional): Number of parallel jobs. 1 forces serial. Defaults to 1.
456
457    Returns:
458        np.ndarray: A NumPy array of objective values.
459    """
460    # Ensure input is a standard numpy array for consistent handling
461    if isinstance(population, list):
462        population = np.array(population)
463
464    # 1. Optimistic Approach: Vectorized Execution
465    # If the user's function supports (N, D) -> (N,) input, this is instant.
466    try:
467        result = function(population)
468        # Verify result is a valid array of the correct shape (N,) or (N, 1)
469        if isinstance(result, np.ndarray) and result.size == population.shape[0]:
470            return result.flatten()
471    except Exception:
472        # Function does not support matrix input, proceed to row-by-row methods
473        pass
474
475    # 2. Serial Execution (User requested np.apply_along_axis)
476    if n_jobs == 1:
477        # Apply function along axis 1 (rows).
478        # Note: apply_along_axis iterates in Python but handles array wrapping cleanly.
479        return np.apply_along_axis(function, 1, population)
480
481    # 3. Parallel Execution (Joblib)
482    else:
483        try:
484            # Backend 'loky' is robust for generic Python objects.
485            obj_list = joblib.Parallel(backend="loky", n_jobs=n_jobs)(
486                joblib.delayed(function)(c) for c in population
487            )
488        except Exception:
489            # Fallback to threading if serialization (pickling) fails
490            obj_list = joblib.Parallel(backend="threading", n_jobs=n_jobs)(
491                joblib.delayed(function)(c) for c in population
492            )
493
494        return np.array(obj_list)

153    def _compute_v(self, dim_idx):
154        """Computes direction numbers V for a specific dimension."""
155        BITS = 32
156        
157        # 1. Handle Dimension 1 (Special Case: Van der Corput)
158        if dim_idx == 0:
159            V = np.zeros(BITS + 1, dtype=np.uint32)
160            for i in range(1, BITS + 1):
161                V[i] = 1 << (BITS - i)
162            return V
163
164        # 2. Handle Dimensions 2+
165        if dim_idx > len(self._DIRECTION_NUMBERS):
166            raise ValueError(f"Max dimension supported is {len(self._DIRECTION_NUMBERS)+1}")
167
168        params = self._DIRECTION_NUMBERS[dim_idx - 1]
169        s = params[1]
170        a = params[2]
171        m = [0] + params[3]
172
173        V = np.zeros(BITS + 1, dtype=np.uint32)
174
175        # Initialize first 's' numbers
176        for i in range(1, s + 1):
177            V[i] = m[i] << (BITS - i)
178
179        # Recurrence for remaining bits
180        for i in range(s + 1, BITS + 1):
181            v_new = V[i - s] ^ (V[i - s] >> s)
182            for k in range(1, s):
183                if (a >> (s - 1 - k)) & 1:
184                    v_new ^= V[i - k]
185            V[i] = v_new
186        return V

193
194        BITS = 32

350def compute_crowding_distance(samples: np.ndarray) -> np.ndarray:
351    """
352    Computes the Crowding Distance (NSGA-II style).
353    Estimates the density of points around each sample.
354    
355    Complexity: O(D * N log N)
356    
357    Args:
358        samples (np.ndarray): Shape (N, D)
359        
360    Returns:
361        np.ndarray: Shape (N,). Higher value = More isolated (Better).
362    """
363    N, D = samples.shape
364    if N == 0:
365        return np.array([])
366
367    distances = np.zeros(N)
368
369    # We compute distance dimension by dimension
370    for d in range(D):
371        # 1. Sort by the current dimension
372        # argsort gives us the indices that would sort the array
373        sorted_indices = np.argsort(samples[:, d])
374        sorted_samples = samples[sorted_indices, d]
375
376        # 2. Handle Boundaries
377        # The min and max points in each dim are always "infinite" distance (most isolated)
378        distances[sorted_indices[0]] = np.inf
379        distances[sorted_indices[-1]] = np.inf
380
381        # 3. Compute Distance for Internal Points
382        # Formula: (Next_Val - Prev_Val) / (Max_Val - Min_Val)
383        scale = sorted_samples[-1] - sorted_samples[0]
384        
385        if scale == 0:
386            continue  # All points are identical in this dimension
387
388        # Vectorized difference: P[i+1] - P[i-1]
389        # We slice sorted_samples from [2:] and [:-2] to get next/prev neighbors
390        dim_dist = (sorted_samples[2:] - sorted_samples[:-2]) / scale
391        
392        # Add to the cumulative score of the corresponding original indices
393        # Indices [1:-1] correspond to the internal points we just computed
394        distances[sorted_indices[1:-1]] += dim_dist
395
396    # Replace infinite values (boundaries) with the max finite value found 
397    # so probabilities don't break
398    max_finite = np.max(distances[np.isfinite(distances)])
399    distances[np.isinf(distances)] = max_finite * 2.0  # Give them a boost
400
401    return distances

404def score_2_probs(scores: np.ndarray, temperature: float = 1.0) -> np.ndarray:
405    """
406    Converts minimization scores into a probability distribution.
407    Uses Min-Max normalization to handle negative values robustly.
408
409    Args:
410        scores (np.ndarray): Objective values (lower is better).
411        temperature (float): Controls selection pressure (optional). 
412                             >1.0 makes distribution flatter (more random).
413                             <1.0 makes distribution sharper (greedy).
414
415    Returns:
416        np.ndarray: Probabilities summing to 1.0.
417    """
418    # 1. Handle Constant Scores (Variance is 0)
419    if np.all(scores == scores[0]):
420        return np.ones_like(scores) / len(scores)
421
422    # 2. Min-Max Normalization -> Maps to [0, 1]
423    # S_norm = (S - min) / (max - min)
424    min_s = np.min(scores)
425    max_s = np.max(scores)
426    norm_scores, _, _ = scale(scores, min_s, max_s)
427
428    # 3. Invert (Minimization: Low score -> High prob)
429    # 0.0 (Best) becomes 1.0
430    # 1.0 (Worst) becomes 0.0
431    # We add a small epsilon to avoid 0.0 probability for the worst candidate
432    weights = (1.0 - norm_scores) + 1e-6
433
434    # 4. Optional: Temperature Scaling (Weights ^ (1/T))
435    if temperature != 1.0:
436        weights = weights ** (1.0 / temperature)
437
438    # 5. Normalize to Probabilities
439    return weights / np.sum(weights)

462    if isinstance(population, list):
463        population = np.array(population)

467    try:
468        result = function(population)
469        # Verify result is a valid array of the correct shape (N,) or (N, 1)

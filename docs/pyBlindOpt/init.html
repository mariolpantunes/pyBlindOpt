 29def get_initial_population(
 30    n_pop: int, bounds: np.ndarray, sampler: utils.Sampler
 31) -> np.ndarray:
 32    """
 33    Generates a population matrix using a specified Sampler.
 34
 35    Args:
 36        n_pop (int): Number of individuals.
 37        bounds (np.ndarray): Search space bounds.
 38        sampler (utils.Sampler): The sampling strategy (Random, LHS, Sobol, etc.).
 39
 40    Returns:
 41        np.ndarray: Population matrix $(N, D)$.
 42    """
 43    return sampler.sample(n_pop, bounds)

 46def _parse_population_arg(
 47    population: np.ndarray | utils.Sampler | None,
 48    n_pop: int,
 49    bounds: np.ndarray,
 50    rng: np.random.Generator,
 51) -> tuple[np.ndarray, int]:
 52    """
 53    Standardizes the population argument handling.
 54
 55    Converts user input (None, Array, or Sampler) into a concrete population array.
 56    """
 57    if isinstance(population, utils.Sampler):
 58        pop = get_initial_population(n_pop, bounds, population)
 59    elif isinstance(population, np.ndarray):
 60        pop = utils.check_bounds(population, bounds)
 61        n_pop = pop.shape[0]  # Update n_pop to match provided array
 62    elif population is None:
 63        sampler = utils.RandomSampler(rng)
 64        pop = get_initial_population(n_pop, bounds, sampler)
 65    else:
 66        raise ValueError("Population must be None, ndarray, or PopulationSampler.")
 67
 68    return pop, n_pop

 71def opposition_based(
 72    objective: collections.abc.Callable,
 73    bounds: np.ndarray,
 74    population: np.ndarray | utils.Sampler | None = None,
 75    n_pop: int = 10,
 76    n_jobs: int = 1,
 77    seed: int | np.random.Generator | None = 42,
 78) -> np.ndarray:
 79    """
 80    Opposition-Based Learning (OBL) Initialization.
 81
 82    Generates a population and its "opposite" in the search space, then selects
 83    the best $N$ individuals from the combined pool ($2N$).
 84
 85    **Opposite Point Formula:**
 86    For a point $x \\in [a, b]$, the opposite $\\breve{x}$ is:
 87    $$ \\breve{x} = a + b - x $$
 88
 89    **Analogy:**
 90    If looking for gold, check your current spot, but also check the exact
 91    opposite side of the map. Often, if one side is bad, the opposite is promising.
 92
 93    Returns:
 94        np.ndarray: The fittest $N$ individuals from the union of random and opposite populations.
 95    """
 96    rng = (
 97        np.random.default_rng(seed)
 98        if not isinstance(seed, np.random.Generator)
 99        else seed
100    )
101
102    pop, n_pop = _parse_population_arg(population, n_pop, bounds, rng)
103
104    # compute the fitness of the initial population
105    scores = utils.compute_objective(pop, objective, n_jobs)
106
107    # compute the opposition population
108    lower = bounds[:, 0]
109    upper = bounds[:, 1]
110    pop_opp = utils.check_bounds(lower + upper - pop, bounds)
111
112    # compute the fitness of the opposition population
113    scores_opp = utils.compute_objective(pop_opp, objective, n_jobs)
114
115    # merge the results and filter
116    combined_pop = np.vstack((pop, pop_opp))
117    combined_scores = np.concatenate((scores, scores_opp))
118    top_k_indices = np.argpartition(combined_scores, n_pop)[:n_pop]
119
120    return combined_pop[top_k_indices]

123def round_init(
124    objective: collections.abc.Callable,
125    bounds: np.ndarray,
126    sampler: utils.Sampler,
127    n_pop: int = 10,
128    n_rounds: int = 3,
129    diversity_weight: float = 0.5,
130    n_jobs: int = 1,
131) -> np.ndarray:
132    """
133    Tournament-like Initialization.
134
135    Samples a larger pool ($N \\times \\text{rounds}$), evaluates them, and selects
136    the final $N$ based on a weighted combination of Fitness and Diversity (Crowding Distance).
137
138    **Selection Probability:**
139    $$ P(x) \\propto (1 - w) \\cdot P_{fitness}(x) + w \\cdot P_{diversity}(x) $$
140
141    Args:
142        n_rounds (int): Multiplier for pool size.
143        diversity_weight (float): Trade-off between quality (0.0) and spread (1.0).
144
145    Returns:
146        np.ndarray: Selected population.
147    """
148    total_candidates = n_pop * n_rounds
149    full_pool = sampler.sample(total_candidates, bounds)
150
151    fitness = np.zeros(total_candidates)
152    for i in range(0, total_candidates, n_pop):
153        batch = full_pool[i : i + n_pop]
154        fitness[i : i + n_pop] = utils.compute_objective(batch, objective, n_jobs)
155
156    prob_fitness = utils.score_2_probs(fitness)
157
158    if diversity_weight > 0:
159        crowding = utils.compute_crowding_distance(full_pool)
160        prob_dist = utils.score_2_probs(-crowding)
161    else:
162        prob_dist = np.zeros_like(prob_fitness)
163
164    final_probs = (1.0 - diversity_weight) * prob_fitness + diversity_weight * prob_dist
165    # Normalize (Floating point math might make sum slightly != 1.0)
166    final_probs /= np.sum(final_probs)
167
168    selected_indices = sampler.rng.choice(
169        total_candidates, size=n_pop, replace=False, p=final_probs
170    )
171
172    return full_pool[selected_indices]

175def oblesa(
176    objective: collections.abc.Callable,
177    bounds: np.ndarray,
178    population: np.ndarray | utils.Sampler | None = None,
179    n_pop: int = 10,
180    n_jobs: int = 1,
181    epochs: int = 1024,
182    lr: float = 0.01,
183    search_mode: str = "radius",
184    k: int | None = None,
185    decay: float = 0.9,
186    batch_size: int = 50,
187    tol: float = 1e-3,
188    radius: float | None = None,
189    border_strategy: str = "repulsive",
190    metric: str | collections.abc.Callable = "softened_inverse",
191    seed: int | np.random.Generator | None = 42,
192    **metric_kwargs,
193) -> np.ndarray:
194    """
195    OBLESA (Opposition-Based Learning with Empty Space Search) Initialization.
196
197    Combines OBL with Empty Space Search (`ess.esa`) to ensure
198    the population is not only high-quality but also maximally distributed
199    (low potential energy configuration).
200
201    Returns:
202        np.ndarray: Optimized population.
203    """
204    rng = (
205        np.random.default_rng(seed)
206        if not isinstance(seed, np.random.Generator)
207        else seed
208    )
209
210    ran_pop, n_pop = _parse_population_arg(population, n_pop, bounds, rng)
211
212    lower, upper = bounds[:, 0], bounds[:, 1]
213    opp_pop = utils.check_bounds(lower + upper - ran_pop, bounds)
214
215    combined_samples = np.vstack((ran_pop, opp_pop))
216    emp_pop = ess.esa(
217        combined_samples,
218        bounds,
219        n=2 * n_pop,
220        epochs=epochs,
221        lr=lr,
222        k=k,
223        decay=decay,
224        batch_size=batch_size,
225        radius=radius,
226        search_mode=search_mode,
227        border_strategy=border_strategy,
228        tol=tol,
229        seed=rng,
230        metric=metric,
231        **metric_kwargs,
232    )
233
234    population = np.vstack((ran_pop, opp_pop, emp_pop))
235    scores = np.zeros(population.shape[0])
236
237    for i in range(0, population.shape[0], n_pop):
238        end = min(i + n_pop, population.shape[0])
239        batch = population[i:end]
240        scores[i:end] = utils.compute_objective(batch, objective, n_jobs)
241
242    probs = utils.score_2_probs(scores)
243
244    try:
245        idx = rng.choice(population.shape[0], size=n_pop, replace=False, p=probs)
246    except ValueError:
247        idx = np.argpartition(scores, n_pop)[:n_pop]
248
249    return population[idx]

252def quasi_opposition_based(
253    objective: collections.abc.Callable,
254    bounds: np.ndarray,
255    population: np.ndarray | utils.Sampler | None = None,
256    n_pop: int = 10,
257    n_jobs: int = 1,
258    seed: int | np.random.Generator | None = 42,
259) -> np.ndarray:
260    """
261    Quasi-Opposition Based Learning (QOBL) Initialization.
262
263    An extension of OBL. Instead of checking the exact opposite point, it samples
264    a random point between the search space center $C$ and the opposite point $\\breve{x}$.
265
266    **Formula:**
267    $$ C = \\frac{a + b}{2}, \\quad \\breve{x} = a + b - x $$
268    $$ x_{q} \\sim U(\\min(C, \\breve{x}), \\max(C, \\breve{x})) $$
269
270    Ref: "A comprehensive study of opposition-based learning" (2014).
271
272    Returns:
273        np.ndarray: The fittest $N$ individuals from the combined pool.
274    """
275    rng = (
276        np.random.default_rng(seed)
277        if not isinstance(seed, np.random.Generator)
278        else seed
279    )
280
281    # 1. Base Population
282    pop, n_pop = _parse_population_arg(population, n_pop, bounds, rng)
283    scores = utils.compute_objective(pop, objective, n_jobs)
284
285    # 2. Compute Center and Opposite
286    lower, upper = bounds[:, 0], bounds[:, 1]
287    center = (lower + upper) / 2.0
288
289    # Standard Opposition: x_opp = a + b - x
290    pop_opp = lower + upper - pop
291
292    # 3. Quasi-Opposition Logic
293    # We sample uniformly between [Center, Opposite]
294    # Note: center and pop_opp are arrays (N, D).
295    # We need element-wise min/max to ensure correct random range
296    low_bound = np.minimum(center, pop_opp)
297    high_bound = np.maximum(center, pop_opp)
298
299    pop_quasi = rng.uniform(low_bound, high_bound)
300
301    # Check bounds (QOBL can sometimes drift slightly, though theoretically safe here)
302    pop_quasi = utils.check_bounds(pop_quasi, bounds)
303
304    # 4. Evaluate
305    scores_quasi = utils.compute_objective(pop_quasi, objective, n_jobs)
306
307    # 5. Selection (Greedy vs Combined)
308    # Standard QOBL usually merges and picks top N
309    combined_pop = np.vstack((pop, pop_quasi))
310    combined_scores = np.concatenate((scores, scores_quasi))
311
312    top_k_indices = np.argpartition(combined_scores, n_pop)[:n_pop]
313
314    return combined_pop[top_k_indices]

Quasi-Opposition Based Learning (QOBL) Initialization.

An extension of OBL. Instead of checking the exact opposite point, it samples a random point between the search space center $C$ and the opposite point $\breve{x}$.

Formula: $$ C = \frac{a + b}{2}, \quad \breve{x} = a + b - x $$ $$ x_{q} \sim U(\min(C, \breve{x}), \max(C, \breve{x})) $$

Ref: "A comprehensive study of opposition-based learning" (2014).

Returns:

np.ndarray: The fittest $N$ individuals from the combined pool.
